```python
class AirConditioningLoad(FlexibleLoad):
    def __init__(self, max_power, min_power, response_time, ramp_rate, comfort_temp):
        super().__init__(max_power, min_power, response_time, ramp_rate)
        
        # === 原有属性 ===
        self.comfort_temp = comfort_temp  # 用户舒适温度（℃）
        self.current_temp = 25.0          # 当前室内温度 T_in（℃）
        
        # === 新增属性：设定温度驱动模型 ===
        self.set_temp = 26.0              # 空调设定温度 T_set（℃）
        self.set_temp_min = 24.0          # 设定温度下限
        self.set_temp_max = 28.0          # 设定温度上限
        
        # 论文公式参数
        self.k_set = 0.0638               # 设定温度调节系数
        self.k_in = 0.0627                # 室内温度偏差系数
        
        # 记录上一步的设定温度（用于计算 ΔT_set）
        self.prev_set_temp = self.set_temp

    def adjust_set_temperature(self, delta_t_set):
        """调整空调设定温度"""
        self.prev_set_temp = self.set_temp
        self.set_temp = np.clip(
            self.set_temp + delta_t_set,
            self.set_temp_min,
            self.set_temp_max
        )
        actual_delta = self.set_temp - self.prev_set_temp
        return actual_delta
    
    def calculate_target_power_from_temperature(self):
        """根据设定温度和室内温度计算目标功率（论文公式）"""
        # 计算设定温度变化量
        delta_t_set = self.set_temp - self.prev_set_temp
        
        # 计算室内温度偏差
        temp_deviation = self.current_temp - self.set_temp
        
        # 应用论文公式
        target_power = (
            self.current_power                          # 当前功率
            - self.k_set * self.max_power * delta_t_set # 设定温度调节项
            + self.k_in * self.max_power * temp_deviation # 温度偏差补偿项
        )
        
        # 限制在物理合理范围内（空调功率不能为负）
        target_power = np.clip(target_power, 0.0, self.max_power)
        
        return target_power
```
